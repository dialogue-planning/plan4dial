 63@config(auto_simplify=True)
 64def convert_to_formula(condition: Dict):
 65    formula_terms = []
 66    if type(condition) == list:
 67        formula_terms.extend(
 68            [
 69                formula
 70                for nesting in condition
 71                for formula in convert_to_formula(nesting)
 72            ]
 73        )
 74    else:
 75        for connective, config in condition.items():
 76            if connective not in ["and", "or"]:
 77                formula_terms.extend(
 78                    Var(f"({connective}-{base_key}-{base_value})")
 79                    for base_key, base_value in config.items()
 80                )
 81            else:
 82                formula_terms.append(
 83                    And(convert_to_formula(config))
 84                    if connective == "and"
 85                    else Or(convert_to_formula(config))
 86                )
 87    return formula_terms

 96def configure_value_setter(loaded_yaml, ctx_var):
 97    processed = deepcopy(loaded_yaml["actions"])
 98    ctx_var_cfg = loaded_yaml["context_variables"][ctx_var]
 99    var_options = (
100        list(ctx_var_cfg["options"].keys())
101        if type(ctx_var_cfg["options"]) == dict
102        else ctx_var_cfg["options"]
103    )
104
105    for v in var_options:
106        option_value_name = f"{ctx_var}-value-{v.replace(' ', '_')}"
107        loaded_yaml["context_variables"][option_value_name] = {
108            "type": "flag",
109            "init": False,
110        }
111    # whenever we lose knowledge of the context variable, reset all values
112    for act in loaded_yaml["actions"]:
113        for eff, eff_config in loaded_yaml["actions"][act]["effect"].items():
114            for option in eff_config:
115                for out, out_config in eff_config[option]["outcomes"].items():
116                    if "updates" in out_config:
117                        for update_var, update_cfg in out_config["updates"].items():
118                            if ctx_var == update_var and "known" in update_cfg:
119                                if update_cfg["known"] == False:
120                                    for v in var_options:
121                                        option_value_name = f"{ctx_var}-value-{v.replace(' ', '_')}"
122                                        processed[act]["effect"][eff][option]["outcomes"][out]["updates"][option_value_name] = {"value": False}
123    processed[f"set-{ctx_var}"] = {
124        "type": "system",
125        "subtype": "Context dependent determination",
126        "condition": {
127            **{ctx_var: {"known": True}},
128            **{f"{ctx_var}-value-{v.replace(' ', '_')}": {"value": False} for v in var_options},
129        },
130        "effect": {
131            "set-valid-value": {
132                "oneof": {
133                    "outcomes": {
134                        v.replace(' ', '_'): {
135                            "updates": {
136                                f"{ctx_var}-value-{v.replace(' ', '_')}": {"value": True},
137                            },
138                            "context": {ctx_var: {"value": v}},
139                        }
140                        for v in var_options
141                    }
142                }
143            }
144        },
145    }
146    loaded_yaml["actions"] = processed

149def reset_force_in_outcomes(clarify, prior_outcomes, forced_name):
150    # every outcome in the forced action other than the fallback/unclear will undo the force
151    new_outcomes = {}
152    reset_force = False
153    for out, out_config in prior_outcomes.items():
154        if "intent" in out_config:
155            intent = out_config["intent"]
156            if intent != "fallback":
157                if clarify:
158                    if intent != "deny":
159                        reset_force = True
160                else:
161                    if type(intent) == str:
162                        if "maybe" not in intent:
163                            reset_force = True
164                    else:
165                        if "maybe" not in intent.values():
166                            reset_force = True
167        else:
168            reset_force = True
169        if reset_force:
170            out_config["updates"][forced_name] = {"value": False}
171        new_outcomes[out] = out_config
172    return new_outcomes

247def convert_ctx_var(loaded_yaml):
248    processed = deepcopy(loaded_yaml["context_variables"])
249    # convert context variables
250    processed = {var: {} for var in loaded_yaml["context_variables"]}
251    for ctx_var, cfg in loaded_yaml["context_variables"].items():
252        json_ctx_var = {}
253        json_ctx_var["type"] = cfg["type"]
254        if cfg["type"] == "enum":
255            if type(cfg["options"]) == dict:
256                json_ctx_var["config"] = list(cfg["options"].keys())
257            else:
258                json_ctx_var["config"] = cfg["options"]
259        elif cfg["type"] == "flag" or cfg["type"] == "fflag":
260            json_ctx_var["config"] = cfg["init"]
261        else:
262            if "extraction" in cfg:
263                json_ctx_var["config"] = {"extraction": cfg["extraction"]}
264                if "method" in cfg:
265                    json_ctx_var["config"]["method"] = cfg["method"]
266                elif "pattern" in cfg:
267                    json_ctx_var["config"]["pattern"] = cfg["pattern"]
268            else:
269                json_ctx_var["config"] = "null"
270        if "known" in cfg:
271            json_ctx_var["known"] = cfg["known"]
272        processed[ctx_var] = json_ctx_var
273    loaded_yaml["context_variables"] = processed

291def add_follow_ups(loaded_yaml):
292    processed = deepcopy(loaded_yaml["actions"])
293    forced_acts = []
294    for act in loaded_yaml["actions"]:
295        for eff, eff_config in loaded_yaml["actions"][act]["effect"].items():
296            for option in eff_config:
297                outcomes = eff_config[option]["outcomes"]
298                for out, out_config in outcomes.items():
299                    next_outcome = deepcopy(out_config)
300                    if "follow_up" in next_outcome:
301                        forced = next_outcome["follow_up"]
302                        next_outcome["updates"][f"forcing__{forced}"] = {"value": True}
303                        forced_acts.append(forced)
304                    if "response_variants" in next_outcome:
305                        next_outcome["updates"].update(configure_fallback_true())
306                    processed[act]["effect"][eff][option]["outcomes"][
307                        out
308                    ] = next_outcome
309    with_forced = deepcopy(processed)
310    for forced in forced_acts:
311        name = f"forcing__{forced}"
312        clarify_name = f"clarify__{forced}"
313        for act in processed:
314            # don't lock fallback/unclear so we can loop on a forced action if needed
315            if act != forced and act != clarify_name and act != "dialogue_statement":
316                with_forced[act]["condition"][name] = {"value": False}
317        for eff, eff_config in loaded_yaml["actions"][forced]["effect"].items():
318            for option in eff_config:
319                with_forced[forced]["effect"][eff][option][
320                    "outcomes"
321                ] = reset_force_in_outcomes(
322                    False, processed[forced]["effect"][eff][option]["outcomes"], name
323                )
324        if clarify_name in processed:
325            for eff, eff_config in loaded_yaml["actions"][clarify_name][
326                "effect"
327            ].items():
328                for option in eff_config:
329                    with_forced[clarify_name]["effect"][eff][option][
330                        "outcomes"
331                    ] = reset_force_in_outcomes(
332                        True,
333                        processed[clarify_name]["effect"][eff][option]["outcomes"],
334                        name,
335                    )
336        loaded_yaml["context_variables"][name] = {"type": "flag", "init": False}
337    loaded_yaml["actions"] = with_forced

340def duplicate_for_or_condition(loaded_yaml):
341    processed = deepcopy(loaded_yaml["actions"])
342    for act, act_cfg in loaded_yaml["actions"].items():
343        for cond, cond_cfg in act_cfg["condition"].items():
344            # currently only handles "or"s and "or"s of "ands" (when multiple conditions are listed under one bullet)
345            if cond == "or":
346                idx = 1
347                for or_cond in cond_cfg:
348                    new_cond = deepcopy(act_cfg["condition"])
349                    del new_cond["or"]
350                    for k, v in or_cond.items():
351                        new_cond[k] = v
352                    next_act_name = f"{act}-or-{idx}"
353                    processed[next_act_name] = deepcopy(processed[act])
354                    processed[next_act_name]["condition"] = new_cond
355                    idx += 1
356                del processed[act]
357    loaded_yaml["actions"] = processed

359def duplicate_for_or_when_condition(loaded_yaml):
360    processed = deepcopy(loaded_yaml["actions"])
361    for act, act_cfg in loaded_yaml["actions"].items():
362        for eff, eff_config in loaded_yaml["actions"][act]["effect"].items():
363            for option in eff_config:
364                for out, out_config in eff_config[option]["outcomes"].items():
365                    if "updates" in out_config:
366                        for update_var, update_cfg in out_config["updates"].items():
367                            # for now, assume we only use "and" explicitly to stack "when" expressions
368                            if update_var == "and":
369                                for when_expr in update_cfg:
370                                    for when_cond, when_cond_cfg in when_expr["when"]["condition"].items():
371                                        if when_cond == "or":
372                                            for or_cond in when_cond_cfg:
373                                                new_when = deepcopy(when_expr)
374                                                del new_when["when"]["condition"]["or"]
375                                                for k, v in or_cond.items():
376                                                    new_when["when"]["condition"][k] = v
377                                                processed[act]["effect"][eff][option]["outcomes"][out]["updates"][update_var].append(new_when)
378                                    processed[act]["effect"][eff][option]["outcomes"][out]["updates"][update_var].remove(when_expr)
379    loaded_yaml["actions"] = processed

382def add_value_setters(loaded_yaml):
383    processed = deepcopy(loaded_yaml)
384    use_value_setters = set()
385    # stores actions that need a separate action where their values are encoded in PDDL.
386    # necessary whenever an action is contingent on a variable being a certain value.
387    for act, act_cfg in loaded_yaml["actions"].items():
388        for cond, cond_cfg in act_cfg["condition"].items():
389            if "value" in cond_cfg:
390                option = cond_cfg["value"]
391                if type(option) == str:
392                    if option not in loaded_yaml["context_variables"][cond]["options"]:
393                        raise AssertionError(
394                            f'Cannot specify the value "{option}" for the context variable "{cond}".'
395                        )
396                    # if not done already, create new "value-setting" actions, conditions, etc.
397                    if f"{cond}-value-{option.replace(' ', '_')}" not in processed["context_variables"]:
398                        configure_value_setter(processed, cond)
399                        use_value_setters.add(cond)
400                    # reset old condition to the refactored condition so we can specify value
401                    # only delete the value portion (may still have "known")
402                    del processed["actions"][act]["condition"][cond]["value"]
403                    # delete whole condition if empty
404                    if processed["actions"][act]["condition"][cond] == {}:
405                        del processed["actions"][act]["condition"][cond]
406                    # replace
407                    processed["actions"][act]["condition"][f"{cond}-value-{option.replace(' ', '_')}"] = {
408                        "value": True
409                    }
410    # NEED TO FIX ISSUE WHERE ASSIGNING A VARIABLE TO NULL DOES NOT RE-ASSIGN THE VALUES
411    # OF THE "VALUE" CONTEXT VARIABLES. I.E. CUISINE == NULL DOES NOT CHANGE THAT
412    # CUISINE-VALUE-MEXICAN IS STILL TRUE. EASY SOLUTION: AUTO-CREATE A SYSTEM ACTION SUCH THAT
413    # IF WE DO NOT KNOW THE CONTEXT VARIABLE AND ONE OF THE VALUES IS TRUE (OR), RESET ALL VALUES.
414    loaded_yaml["actions"], loaded_yaml["context_variables"] = (
415        processed["actions"],
416        processed["context_variables"],
417    )

420def convert_actions(loaded_yaml):
421    processed = deepcopy(loaded_yaml["actions"])
422    for act in loaded_yaml["actions"]:
423        # convert preconditions
424        json_config_cond = []
425        condition = loaded_yaml["actions"][act]["condition"]
426        for cond, cond_cfg in condition.items():
427            for cond_config_key, cond_config_val in cond_cfg.items():
428                if cond_config_key == "known":
429                    json_config_cond.append(
430                        ([cond, "Known"] if cond_config_val else [cond, "Unknown"])
431                        if type(cond_config_val) == bool
432                        else [cond, "Uncertain"]
433                    )
434                elif cond_config_key == "value":
435                    json_config_cond.append([cond, cond_config_val])
436        processed[act]["condition"] = json_config_cond
437        for eff, eff_config in loaded_yaml["actions"][act]["effect"].items():
438            converted_eff = deepcopy(eff_config)
439            converted_eff["global-outcome-name"] = eff
440            intents = []
441            for option in eff_config:
442                converted_eff["type"] = option
443                outcomes = eff_config[option]["outcomes"]
444                outcomes_list = []
445                for out, out_config in outcomes.items():
446                    next_outcome = deepcopy(out_config)
447                    next_outcome["name"] = f"{act}_DETDUP_{eff}-EQ-{out}"
448                    if "intent" not in out_config:
449                        next_outcome["intent"] = None
450                    else:
451                        intents.append(out_config["intent"])
452                    next_outcome["assignments"] = {}
453                    if "updates" in out_config:
454                        for update_var, update_cfg in out_config["updates"].items():
455                            next_outcome["updates"][update_var]["variable"] = update_var
456                            if "known" in update_cfg:
457                                next_outcome["assignments"][
458                                    f"${update_var}"
459                                ] = configure_assignments(update_cfg["known"])
460                                next_outcome["updates"][update_var][
461                                    "certainty"
462                                ] = configure_certainty(update_cfg["known"])
463                                del next_outcome["updates"][update_var]["known"]
464                            if "value" not in update_cfg:
465                                 next_outcome["updates"][update_var]["value"] = None
466                            next_outcome["updates"][update_var][
467                                "interpretation"
468                            ] = "json"
469                    outcomes_list.append(next_outcome)
470                converted_eff["outcomes"] = outcomes_list
471                del converted_eff[option]
472                processed[act]["effect"] = converted_eff
473        if intents:
474            processed[act]["intents"] = {}
475            for intent in intents:
476                if type(intent) == str:
477                    if intent in loaded_yaml["intents"]:
478                        processed[act]["intents"][intent] = loaded_yaml["intents"][
479                            intent
480                        ]
481    loaded_yaml["actions"] = processed

484def convert_yaml(filename: str):
485    loaded_yaml = yaml.load(open(filename, "r"), Loader=yaml.FullLoader)
486    base_setup(loaded_yaml)
487    instantiate_clarification_actions(loaded_yaml)
488    instantiate_advanced_custom_actions(loaded_yaml)
489    instantiate_effects_add_fallbacks(loaded_yaml)
490    add_follow_ups(loaded_yaml)
491    duplicate_for_or_condition(loaded_yaml)
492    duplicate_for_or_when_condition(loaded_yaml)
493    add_value_setters(loaded_yaml)
494    convert_ctx_var(loaded_yaml)
495    convert_intents(loaded_yaml)
496    convert_actions(loaded_yaml)
497    return loaded_yaml
